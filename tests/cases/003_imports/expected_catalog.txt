:apt:default:
    working_directory: __WORKDIR__/.apt
    environment:
        DOCKER_VERSION: 1.2.3
        EBRO_ROOT: __WORKDIR__
    script: echo 'Installing stuff with apt'
    sources:
        - __WORKDIR__/something
:default:
    working_directory: __WORKDIR__
    environment:
        DOCKER_VERSION: 1.2.3
        EBRO_ROOT: __WORKDIR__
    requires:
        - :docker-config
        - :docker:default
:docker-compose-project:
    working_directory: __WORKDIR__
    environment:
        DOCKER_VERSION: 1.2.3
        EBRO_ROOT: __WORKDIR__
    requires:
        - :docker:compose:default
    script: docker compose up -d
    sources:
        - __WORKDIR__/docker-compose.yml
:docker-config:
    working_directory: __WORKDIR__
    environment:
        DOCKER_VERSION: 1.2.3
        EBRO_ROOT: __WORKDIR__
    required_by:
        - :docker:default
    script: configure docker
:docker:compose:default:
    working_directory: __WORKDIR__/docker-compose
    environment:
        DOCKER_VERSION: 1.2.3
        DOCKER_VERSION_GITHUB_ID: v1.2.3
        EBRO_ROOT: __WORKDIR__
    script: echo 'docker compose'
:docker:default:
    working_directory: __WORKDIR__/docker
    environment:
        DOCKER_VERSION: 1.2.3
        DOCKER_VERSION_GITHUB_ID: v1.2.3
        EBRO_ROOT: __WORKDIR__
    requires:
        - :docker:compose:default
        - :apt:default
